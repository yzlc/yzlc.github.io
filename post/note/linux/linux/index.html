#!/bin/bash
# chkconfig:- 85 15
# description:开机启动
source /etc/profile

# 备份
mkdir -p backup
cp -u *.jar backup/

cd update
for process_name in *.jar; do
  echo "$process_name"
  mv -u "$process_name" ..

  # 获取进程ID
  pid=$(pgrep -f $process_name)

  if [[ -n $pid ]]; then
    # 杀死进程
    echo "kill..."
    kill $pid
    
    # 循环检查进程是否退出
    while ps -p $pid > /dev/null; do
        sleep 1
    done
  fi

  # 启动进程
  echo "start..."
  cd ..
  nohup java -jar -Dfile.encoding=utf8 $process_name >/dev/null 2>&1 &
  cd update

  # 检查
  pid=$(pgrep -f $process_name)

  if [[ -n $pid ]]; then
    echo "success"
  else
    echo "fail"
  fi

done
